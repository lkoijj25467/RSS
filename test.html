def WKNN(K, rssi_fgpt, rssi_dm, rssi):
    t1, t2 = rssi_fgpt.shape
    t3 = np.tile(rssi, (t1, 1))  # 
    temp1 = rssi_fgpt - t3
    wknn = np.sqrt(np.sum((temp1 ** 2), axis=1))
    LMAX = np.max(wknn)
    
    wknnfmt = np.zeros((K, 3))  # 
    xwknn = 0  # WKNN
    ywknn = 0
    wknnsum = 0
    
    # Kwknnfmt
    for k in range(K):
        L = np.min(wknn)
        M = np.argmin(wknn)
        wknnfmt[k, 0] = rssi_dm[M, 0]
        wknnfmt[k, 1] = rssi_dm[M, 1]
        wknnfmt[k, 2] = L
        wknn[M] = LMAX
        wknnsum += 1 / L
    
    # 
    for k in range(K):
        xwknn += (1 / wknnfmt[k, 2]) / wknnsum * wknnfmt[k, 0]
        ywknn += (1 / wknnfmt[k, 2]) / wknnsum * wknnfmt[k, 1]
    
    xknn = 0  # KNN
    yknn = 0
    
    # KNN
    for k in range(K):
        xknn += wknnfmt[k, 0]
        yknn += wknnfmt[k, 1]
    
    xknn /= K
    yknn /= K
    
    return xwknn, ywknn, xknn, yknn
